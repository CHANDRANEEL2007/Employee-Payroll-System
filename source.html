Payroll Formula

perDaySalary = basicSalary / workingDays
absentDays = workingDays - daysPresent
attendanceDeduction = absentDays * perDaySalary
overtimePay = overtimeHours * overtimeRate
grossSalary = basicSalary + hra + da + bonus + overtimePay
totalDeduction = attendanceDeduction + pf + tax + insurance
netSalary = grossSalary - totalDeduction
